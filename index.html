async function loadSeries(seriesId){ const fredUrl = `https://api.stlouisfed.org/fred/series/observations?series_id=${seriesId}&api_key=${apiKey}&file_type=json` const proxyUrl = `https://api.codetabs.com/v1/proxy?quest=${fredUrl}` const res = await fetch(proxyUrl) const json = await res.json() const obs = json.observations.filter(d => d.value !== ".") const dates = obs.map(d => d.date) const values = obs.map(d => parseFloat(d.value)) const latest = values[values.length-1] document.getElementById("val_"+seriesId).innerText = "Latest: " + latest new Chart( document.getElementById("chart_"+seriesId), { type:"line", data:{ labels:dates, datasets:[{ data:values, borderWidth:2, pointRadius:0 }] }, options:{ responsive:true, plugins:{legend:{display:false}}, scales:{x:{display:false}} } } ) }
